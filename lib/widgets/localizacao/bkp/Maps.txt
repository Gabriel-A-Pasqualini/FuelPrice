
import 'package:fuelprice/helper/mapa_localizacao_helper.dart';
import 'package:flutter/material.dart';
import 'package:geolocator/geolocator.dart';


class MapsWidget extends StatefulWidget {
  const MapsWidget({super.key});

  @override
  State<MapsWidget> createState() => _MapsWidgetState();
}

class _MapsWidgetState extends State<MapsWidget> {
  Position? _position;
  String? _erro;
  bool _loading = false;

  Future<void> _obterLocalizacao() async {
    setState(() {
      _loading = true;
      _erro = null;
    });

    try {
      bool serviceEnabled = await Geolocator.isLocationServiceEnabled();
      if (!serviceEnabled) {
        throw 'Servi√ßo de localiza√ß√£o desativado.';
      }

      LocationPermission permission = await Geolocator.checkPermission();
      if (permission == LocationPermission.denied) {
        permission = await Geolocator.requestPermission();
        if (permission == LocationPermission.denied) {
          throw 'Permiss√£o de localiza√ß√£o negada.';
        }
      }

      if (permission == LocationPermission.deniedForever) {
        throw 'Permiss√£o de localiza√ß√£o negada permanentemente.';
      }

      final position = await Geolocator.getCurrentPosition(
        desiredAccuracy: LocationAccuracy.high,
      );

      setState(() => _position = position);
    } catch (e) {
      setState(() => _erro = e.toString());
    } finally {
      setState(() => _loading = false);
    }
  }

  Widget _cardTitulo(String titulo, IconData icone) {
    return Row(
      children: [
        Icon(icone, color: Colors.green[900]),
        const SizedBox(width: 8),
        Text(
          titulo,
          style: TextStyle(
            fontSize: 18,
            fontWeight: FontWeight.bold,
            color: Colors.green[900],
          ),
        ),
      ],
    );
  }

  @override
  Widget build(BuildContext context) {
    final appColor = Colors.green[900];

    return Scaffold(
      appBar: AppBar(
        backgroundColor: appColor,
        title: const Text(
          "Localiza√ß√£o Atual",
          style: TextStyle(
            fontWeight: FontWeight.bold,
            color: Colors.white,
          ),
        ),
        centerTitle: true,
      ),
      body: Padding(
        padding: const EdgeInsets.all(16),
        child: ListView(
          children: [
            Card(
              elevation: 3,
              shape: RoundedRectangleBorder(
                borderRadius: BorderRadius.circular(16),
              ),
              child: Padding(
                padding: const EdgeInsets.all(16),
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: [
                    _cardTitulo("Dados de Localiza√ß√£o", Icons.location_on),
                    const SizedBox(height: 12),

                    if (_loading)
                      const Center(child: CircularProgressIndicator()),

                    if (_erro != null)
                      Text(
                        _erro!,
                        style: const TextStyle(
                          color: Colors.red,
                          fontSize: 15,
                        ),
                      ),

                    if (_position != null) ...[
                      Text(
                        "Latitude: ${_position!.latitude.toStringAsFixed(6)}",
                        style: const TextStyle(fontSize: 16),
                      ),
                      Text(
                        "Longitude: ${_position!.longitude.toStringAsFixed(6)}",
                        style: const TextStyle(fontSize: 16),
                      ),
                      Text(
                        "Precis√£o: ${_position!.accuracy.toStringAsFixed(1)} m",
                        style: const TextStyle(fontSize: 16),
                      ),    

                      const SizedBox(height: 12),

                      mapaLocalizacao(
                        _position!.latitude,
                        _position!.longitude,
                      ),                    
                    ],                  
                  ],
                ),
              ),
            ),

            const SizedBox(height: 20),

            ElevatedButton.icon(
              onPressed: _loading ? null : _obterLocalizacao,
              icon: const Icon(Icons.my_location, color: Colors.white),
              label: const Text(
                "Obter localiza√ß√£o",
                style: TextStyle(fontSize: 18, color: Colors.white),
              ),
              style: ElevatedButton.styleFrom(
                backgroundColor: appColor,
                padding: const EdgeInsets.symmetric(vertical: 14),
                shape: RoundedRectangleBorder(
                  borderRadius: BorderRadius.circular(12),
                ),
              ),
            ),

            const SizedBox(height: 20),

            Divider(color: Colors.grey[400]),

            Text(
              "üìç A localiza√ß√£o √© utilizada apenas para fins informativos.",
              textAlign: TextAlign.center,
              style: TextStyle(
                fontSize: 13,
                color: Colors.grey[700],
                fontStyle: FontStyle.italic,
              ),
            ),
          ],
        ),
      ),
    );
  }
}
